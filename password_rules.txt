# Password Rules Configuration
# Each rule is on one line with format: rule_name:requirement_description:check_function
# Loaded and evaluated at runtime

# Rule 1: Minimum length
min_length:8:len({password}) >= 8

# Rule 2: Must contain uppercase letter
has_uppercase:At least one uppercase letter (A-Z):any(c.isupper() for c in {password})

# Rule 3: Must contain lowercase letter
has_lowercase:At least one lowercase letter (a-z):any(c.islower() for c in {password})

# Rule 4: Must contain digit
has_digit:At least one digit (0-9):any(c.isdigit() for c in {password})

# Rule 5: Must contain special character
has_special:At least one special character (!@#$%^&*):any(c in "!@#$%^&*-_+=." for c in {password})
